JavaScript Array Methods and Object Example

Note: This example demonstrates JavaScript's powerful array methods like map, filter, and reduce, as well as object manipulation and async operations. 
const numbers = [1, 2, 3, 4, 5, 6];
const squares = numbers.map(element => Math.pow(element, 2));
const cubes = numbers.map(element => Math.pow(element, 3));

console.log("Squares:", squares);
console.log("Cubes:", cubes);

const evenNumbers = cubes.filter(element => element % 2 === 0);
const oddNumbers = cubes.filter(element => element % 2 !== 0);

console.log("Even Cubes:", evenNumbers);
console.log("Odd Cubes:", oddNumbers);

const totalSum = numbers.reduce((accumulator, element) => accumulator + element, 0);
console.log("Total Sum:", totalSum);

const ages = [16, 17, 18, 19, 20, 35, 60];
const adults = ages.filter(age => age >= 18);
const children = ages.filter(age => age < 18);

console.log("Adults:", adults);
console.log("Children:", children);

const person = {
    firstName: "Tony",
    lastName: "Popakorn",
    age: 23,
    isEmployee: true,
    work() {
        console.log("Working at TCS");
    },
    eat() {
        console.log("Not eating right now");
    },
    travel: () => {
        console.log("Not traveling at the moment");
    }
};

console.log("Person Details:");
console.log("First Name:", person.firstName);
console.log("Last Name:", person.lastName);
console.log("Age:", person.age);
console.log("Is Employee:", person.isEmployee);
person.work();
person.eat();
person.travel();
